#include <stdio.h>
char a[100][82];
char t[10];
int main()
{
	int n;
	scanf("%d",&n);
	char other[100];
	for (int i=0; i<n;i++){
		scanf("%s",a[i]);
		gets(other);
	}

	for (int i=0;i<n;i++){
		for (int j=i+1;j<n;j++){
			if (strcmp( a[i],a[j])>0){
				strcpy(t,a[i]);
				strcpy(a[i],a[j]);
				strcpy(a[j],t);
			}	
		}
	}
	for(int i=0;i<n;i++){
		printf("%s\n",a[i]);
	}
}
